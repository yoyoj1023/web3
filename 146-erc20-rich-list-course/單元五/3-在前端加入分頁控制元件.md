#### 主要分頁導航元件

建立 `app/rich-list/_components/AdvancedPagination.tsx`：

```tsx
import { ChevronLeftIcon, ChevronRightIcon } from '@heroicons/react/24/outline';

interface AdvancedPaginationProps {
  currentPage: number;
  totalPages: number;
  totalItems: number;
  itemsPerPage: number;
  startItem: number;
  endItem: number;
  hasNextPage: boolean;
  hasPrevPage: boolean;
  onPageChange: (page: number) => void;
  onItemsPerPageChange: (count: number) => void;
  getPageRange: (maxVisible?: number) => number[];
  className?: string;
}

export function AdvancedPagination({
  currentPage,
  totalPages,
  totalItems,
  itemsPerPage,
  startItem,
  endItem,
  hasNextPage,
  hasPrevPage,
  onPageChange,
  onItemsPerPageChange,
  getPageRange,
  className = "",
}: AdvancedPaginationProps) {
  
  const pageRange = getPageRange(7);
  
  // 快速跳轉頁面
  const quickJumpPages = [
    Math.ceil(totalPages * 0.25),
    Math.ceil(totalPages * 0.5),
    Math.ceil(totalPages * 0.75),
  ].filter((page, index, arr) => 
    page > 0 && page <= totalPages && arr.indexOf(page) === index
  );

  const handleQuickJump = () => {
    const page = prompt(`請輸入頁碼 (1-${totalPages}):`);
    const pageNum = parseInt(page || '');
    
    if (!isNaN(pageNum) && pageNum >= 1 && pageNum <= totalPages) {
      onPageChange(pageNum);
    } else if (page) {
      alert('請輸入有效的頁碼');
    }
  };

  return (
    <div className={`bg-white rounded-xl shadow-lg border border-gray-200 ${className}`}>
      {/* 統計資訊和設定區域 */}
      <div className="px-6 py-4 border-b border-gray-200">
        <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-4">
          {/* 統計資訊 */}
          <div className="flex flex-col sm:flex-row items-start sm:items-center gap-2 text-sm text-gray-600">
            <span className="font-medium">
              顯示第 <span className="text-blue-600 font-semibold">{startItem}</span> 至{' '}
              <span className="text-blue-600 font-semibold">{endItem}</span> 項
            </span>
            <span className="hidden sm:inline">•</span>
            <span>
              共 <span className="text-gray-900 font-semibold">{totalItems.toLocaleString()}</span> 項
            </span>
            <span className="hidden sm:inline">•</span>
            <span>
              第 <span className="text-blue-600 font-semibold">{currentPage}</span> 頁，
              共 <span className="text-gray-900 font-semibold">{totalPages}</span> 頁
            </span>
          </div>

          {/* 每頁項目數設定 */}
          <div className="flex items-center gap-3">
            <label className="text-sm font-medium text-gray-700">每頁顯示:</label>
            <select
              value={itemsPerPage}
              onChange={(e) => onItemsPerPageChange(Number(e.target.value))}
              className="
                px-3 py-1.5 text-sm border border-gray-300 rounded-lg
                focus:ring-2 focus:ring-blue-500 focus:border-blue-500
                bg-white hover:bg-gray-50 transition-colors
              "
            >
              <option value={10}>10 項</option>
              <option value={20}>20 項</option>
              <option value={50}>50 項</option>
              <option value={100}>100 項</option>
            </select>
          </div>
        </div>
      </div>

      {/* 分頁控制區域 */}
      <div className="px-6 py-4">
        <div className="flex flex-col lg:flex-row justify-between items-center gap-4">
          
          {/* 快速跳轉區域 */}
          <div className="flex items-center gap-2 text-sm">
            <span className="text-gray-600">快速跳轉:</span>
            {quickJumpPages.map(page => (
              <button
                key={page}
                onClick={() => onPageChange(page)}
                disabled={page === currentPage}
                className="
                  px-2 py-1 text-xs rounded-md
                  bg-gray-100 hover:bg-gray-200
                  disabled:bg-blue-100 disabled:text-blue-600
                  transition-colors
                "
              >
                {Math.round((page / totalPages) * 100)}%
              </button>
            ))}
            <button
              onClick={handleQuickJump}
              className="
                px-2 py-1 text-xs rounded-md
                bg-purple-100 text-purple-600 hover:bg-purple-200
                transition-colors
              "
            >
              跳轉
            </button>
          </div>

          {/* 主要分頁導航 */}
          <div className="flex items-center gap-1">
            {/* 上一頁按鈕 */}
            <button
              onClick={() => onPageChange(currentPage - 1)}
              disabled={!hasPrevPage}
              className="
                flex items-center gap-1 px-3 py-2 text-sm font-medium
                text-gray-700 bg-white border border-gray-300 rounded-l-lg
                hover:bg-gray-50 hover:text-gray-900
                disabled:opacity-50 disabled:cursor-not-allowed
                transition-all duration-200
              "
              aria-label="上一頁"
            >
              <ChevronLeftIcon className="w-4 h-4" />
              <span className="hidden sm:inline">上一頁</span>
            </button>

            {/* 頁碼按鈕 */}
            <div className="flex">
              {pageRange.map((page, index) => (
                <button
                  key={index}
                  onClick={() => onPageChange(page)}
                  disabled={page === currentPage}
                  className={`
                    px-3 py-2 text-sm font-medium border-t border-b border-r border-gray-300
                    transition-all duration-200
                    ${page === currentPage
                      ? 'bg-blue-600 text-white border-blue-600 z-10 shadow-sm'
                      : 'bg-white text-gray-700 hover:bg-gray-50 hover:text-gray-900'
                    }
                  `}
                  aria-label={`第 ${page} 頁`}
                  aria-current={page === currentPage ? 'page' : undefined}
                >
                  {page}
                </button>
              ))}
            </div>

            {/* 下一頁按鈕 */}
            <button
              onClick={() => onPageChange(currentPage + 1)}
              disabled={!hasNextPage}
              className="
                flex items-center gap-1 px-3 py-2 text-sm font-medium
                text-gray-700 bg-white border border-gray-300 rounded-r-lg
                hover:bg-gray-50 hover:text-gray-900
                disabled:opacity-50 disabled:cursor-not-allowed
                transition-all duration-200
              "
              aria-label="下一頁"
            >
              <span className="hidden sm:inline">下一頁</span>
              <ChevronRightIcon className="w-4 h-4" />
            </button>
          </div>

          {/* 進度指示器 */}
          <div className="flex items-center gap-2 text-sm text-gray-600">
            <span>進度:</span>
            <div className="w-24 bg-gray-200 rounded-full h-2">
              <div 
                className="bg-gradient-to-r from-blue-500 to-purple-600 h-2 rounded-full transition-all duration-300"
                style={{ width: `${(currentPage / totalPages) * 100}%` }}
              />
            </div>
            <span>{Math.round((currentPage / totalPages) * 100)}%</span>
          </div>
        </div>
      </div>
    </div>
  );
}
```

#### 搜尋和排序控制元件

建立 `app/rich-list/_components/SearchAndSortControls.tsx`：

```tsx
import { useState, useRef, useEffect } from 'react';
import { 
  MagnifyingGlassIcon, 
  AdjustmentsHorizontalIcon,
  XMarkIcon,
  ArrowUpIcon,
  ArrowDownIcon
} from '@heroicons/react/24/outline';

interface SearchAndSortControlsProps {
  searchQuery: string;
  sortField: string;
  sortDirection: 'asc' | 'desc';
  totalResults: number;
  onSearch: (query: string) => void;
  onSort: (field: string, direction: 'asc' | 'desc') => void;
  onClearFilters: () => void;
  searchSuggestions?: string[];
  isLoading?: boolean;
  className?: string;
}

export function SearchAndSortControls({
  searchQuery,
  sortField,
  sortDirection,
  totalResults,
  onSearch,
  onSort,
  onClearFilters,
  searchSuggestions = [],
  isLoading = false,
  className = "",
}: SearchAndSortControlsProps) {
  
  const [localSearchQuery, setLocalSearchQuery] = useState(searchQuery);
  const [showSuggestions, setShowSuggestions] = useState(false);
  const [showFilters, setShowFilters] = useState(false);
  const searchInputRef = useRef<HTMLInputElement>(null);
  const suggestionsRef = useRef<HTMLDivElement>(null);

  // 同步外部搜尋查詢
  useEffect(() => {
    setLocalSearchQuery(searchQuery);
  }, [searchQuery]);

  // 處理搜尋輸入
  const handleSearchInput = (value: string) => {
    setLocalSearchQuery(value);
    setShowSuggestions(value.length > 0 && searchSuggestions.length > 0);
  };

  // 執行搜尋
  const executeSearch = (query: string = localSearchQuery) => {
    onSearch(query);
    setShowSuggestions(false);
    searchInputRef.current?.blur();
  };

  // 處理鍵盤事件
  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      executeSearch();
    } else if (e.key === 'Escape') {
      setShowSuggestions(false);
      searchInputRef.current?.blur();
    }
  };

  // 排序選項
  const sortOptions = [
    { field: 'balanceNumber', label: '餘額', icon: '💰' },
    { field: 'percentage', label: '佔比', icon: '📊' },
    { field: 'address', label: '地址', icon: '🔗' },
    { field: 'rank', label: '排名', icon: '🏆' },
  ];

  // 點擊外部關閉建議
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (
        suggestionsRef.current && 
        !suggestionsRef.current.contains(event.target as Node) &&
        !searchInputRef.current?.contains(event.target as Node)
      ) {
        setShowSuggestions(false);
      }
    };

    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  return (
    <div className={`bg-white rounded-xl shadow-lg border border-gray-200 ${className}`}>
      <div className="p-6">
        {/* 搜尋區域 */}
        <div className="flex flex-col lg:flex-row gap-4 mb-4">
          {/* 搜尋輸入框 */}
          <div className="relative flex-1">
            <div className="relative">
              <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none">
                <MagnifyingGlassIcon className="h-5 w-5 text-gray-400" />
              </div>
              <input
                ref={searchInputRef}
                type="text"
                value={localSearchQuery}
                onChange={(e) => handleSearchInput(e.target.value)}
                onKeyDown={handleKeyDown}
                onFocus={() => setShowSuggestions(localSearchQuery.length > 0 && searchSuggestions.length > 0)}
                placeholder="搜尋持有者地址或餘額..."
                disabled={isLoading}
                className="
                  block w-full pl-10 pr-12 py-3 border border-gray-300 rounded-lg
                  focus:ring-2 focus:ring-blue-500 focus:border-blue-500
                  disabled:bg-gray-50 disabled:text-gray-500
                  placeholder-gray-400 text-sm
                "
              />
              {localSearchQuery && (
                <button
                  onClick={() => {
                    setLocalSearchQuery('');
                    onSearch('');
                  }}
                  className="absolute inset-y-0 right-0 pr-3 flex items-center"
                >
                  <XMarkIcon className="h-5 w-5 text-gray-400 hover:text-gray-600" />
                </button>
              )}
            </div>

            {/* 搜尋建議 */}
            {showSuggestions && (
              <div 
                ref={suggestionsRef}
                className="absolute z-10 w-full mt-1 bg-white border border-gray-300 rounded-lg shadow-lg max-h-60 overflow-y-auto"
              >
                {searchSuggestions.map((suggestion, index) => (
                  <button
                    key={index}
                    onClick={() => executeSearch(suggestion)}
                    className="w-full px-4 py-2 text-left text-sm hover:bg-gray-50 first:rounded-t-lg last:rounded-b-lg"
                  >
                    <div className="flex items-center gap-2">
                      <MagnifyingGlassIcon className="h-4 w-4 text-gray-400" />
                      <span className="font-mono">{suggestion}</span>
                    </div>
                  </button>
                ))}
              </div>
            )}
          </div>

          {/* 搜尋和過濾器按鈕 */}
          <div className="flex gap-2">
            <button
              onClick={() => executeSearch()}
              disabled={isLoading}
              className="
                flex items-center gap-2 px-6 py-3 bg-blue-600 text-white rounded-lg
                hover:bg-blue-700 disabled:bg-blue-400 disabled:cursor-not-allowed
                font-medium transition-colors
              "
            >
              <MagnifyingGlassIcon className="h-4 w-4" />
              搜尋
            </button>
            
            <button
              onClick={() => setShowFilters(!showFilters)}
              className={`
                flex items-center gap-2 px-4 py-3 border rounded-lg font-medium transition-colors
                ${showFilters 
                  ? 'bg-purple-50 text-purple-700 border-purple-300' 
                  : 'bg-white text-gray-700 border-gray-300 hover:bg-gray-50'
                }
              `}
            >
              <AdjustmentsHorizontalIcon className="h-4 w-4" />
              篩選
            </button>
            
            {(searchQuery || sortField !== 'balanceNumber' || sortDirection !== 'desc') && (
              <button
                onClick={onClearFilters}
                className="
                  flex items-center gap-2 px-4 py-3 bg-gray-100 text-gray-700 rounded-lg
                  hover:bg-gray-200 font-medium transition-colors
                "
              >
                <XMarkIcon className="h-4 w-4" />
                清除
              </button>
            )}
          </div>
        </div>

        {/* 結果統計 */}
        {searchQuery && (
          <div className="mb-4 p-3 bg-blue-50 rounded-lg">
            <p className="text-sm text-blue-700">
              搜尋 "<span className="font-semibold">{searchQuery}</span>" 找到{' '}
              <span className="font-semibold">{totalResults}</span> 個結果
            </p>
          </div>
        )}

        {/* 進階篩選選項 */}
        {showFilters && (
          <div className="border-t pt-4">
            <h4 className="text-sm font-medium text-gray-900 mb-3">排序選項</h4>
            <div className="grid grid-cols-2 lg:grid-cols-4 gap-2">
              {sortOptions.map((option) => (
                <button
                  key={option.field}
                  onClick={() => onSort(
                    option.field, 
                    sortField === option.field && sortDirection === 'desc' ? 'asc' : 'desc'
                  )}
                  className={`
                    flex items-center justify-between p-3 rounded-lg border text-sm font-medium transition-colors
                    ${sortField === option.field
                      ? 'bg-blue-50 text-blue-700 border-blue-300'
                      : 'bg-white text-gray-700 border-gray-300 hover:bg-gray-50'
                    }
                  `}
                >
                  <div className="flex items-center gap-2">
                    <span>{option.icon}</span>
                    <span>{option.label}</span>
                  </div>
                  {sortField === option.field && (
                    <div className="flex items-center">
                      {sortDirection === 'desc' ? (
                        <ArrowDownIcon className="h-4 w-4" />
                      ) : (
                        <ArrowUpIcon className="h-4 w-4" />
                      )}
                    </div>
                  )}
                </button>
              ))}
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
```

#### 增強的富豪榜表格元件

更新 `app/rich-list/_components/EnhancedRichListTable.tsx`：

```tsx
import { useState } from 'react';
import { Address } from "~~/components/scaffold-alchemy";
import { 
  ChevronUpIcon, 
  ChevronDownIcon,
  EyeIcon,
  DocumentDuplicateIcon,
  ShareIcon
} from '@heroicons/react/24/outline';

interface TokenHolder {
  address: string;
  balance: string;
  balanceFormatted: string;
  balanceNumber: number;
  percentage: number;
  rank?: number;
}

interface EnhancedRichListTableProps {
  holders: TokenHolder[];
  tokenSymbol: string;
  totalSupply?: string;
  currentPage: number;
  itemsPerPage: number;
  sortField: string;
  sortDirection: 'asc' | 'desc';
  onSort: (field: string) => void;
  onHolderClick?: (holder: TokenHolder) => void;
  className?: string;
}

export function EnhancedRichListTable({
  holders,
  tokenSymbol,
  totalSupply,
  currentPage,
  itemsPerPage,
  sortField,
  sortDirection,
  onSort,
  onHolderClick,
  className = "",
}: EnhancedRichListTableProps) {
  
  const [selectedHolder, setSelectedHolder] = useState<string | null>(null);
  const [copiedAddress, setCopiedAddress] = useState<string | null>(null);

  // 複製地址到剪貼簿
  const copyAddress = async (address: string) => {
    try {
      await navigator.clipboard.writeText(address);
      setCopiedAddress(address);
      setTimeout(() => setCopiedAddress(null), 2000);
    } catch (error) {
      console.error('複製失敗:', error);
    }
  };

  // 分享持有者資訊
  const shareHolder = async (holder: TokenHolder) => {
    const shareData = {
      title: `${tokenSymbol} 富豪榜 - 第 ${holder.rank} 名`,
      text: `地址 ${holder.address} 持有 ${holder.balanceFormatted} ${tokenSymbol} (${holder.percentage.toFixed(2)}%)`,
      url: window.location.href,
    };

    try {
      if (navigator.share) {
        await navigator.share(shareData);
      } else {
        await copyAddress(shareData.text);
      }
    } catch (error) {
      console.error('分享失敗:', error);
    }
  };

  // 排序按鈕元件
  const SortButton = ({ field, children }: { field: string; children: React.ReactNode }) => (
    <button
      onClick={() => onSort(field)}
      className="flex items-center gap-1 hover:text-blue-600 transition-colors"
    >
      {children}
      <div className="flex flex-col">
        <ChevronUpIcon 
          className={`w-3 h-3 ${
            sortField === field && sortDirection === 'asc' 
              ? 'text-blue-600' 
              : 'text-gray-400'
          }`} 
        />
        <ChevronDownIcon 
          className={`w-3 h-3 -mt-1 ${
            sortField === field && sortDirection === 'desc' 
              ? 'text-blue-600' 
              : 'text-gray-400'
          }`} 
        />
      </div>
    </button>
  );

  // 計算全域排名
  const getGlobalRank = (index: number) => {
    return (currentPage - 1) * itemsPerPage + index + 1;
  };

  // 獲取排名樣式
  const getRankStyle = (rank: number) => {
    if (rank === 1) return "bg-gradient-to-r from-yellow-400 to-yellow-600 text-white";
    if (rank === 2) return "bg-gradient-to-r from-gray-400 to-gray-600 text-white";
    if (rank === 3) return "bg-gradient-to-r from-amber-600 to-amber-800 text-white";
    if (rank <= 10) return "bg-gradient-to-r from-blue-500 to-blue-700 text-white";
    if (rank <= 50) return "bg-gradient-to-r from-green-500 to-green-700 text-white";
    return "bg-gradient-to-r from-gray-500 to-gray-700 text-white";
  };

  // 獲取排名圖標
  const getRankIcon = (rank: number) => {
    if (rank === 1) return "🥇";
    if (rank === 2) return "🥈";
    if (rank === 3) return "🥉";
    if (rank <= 10) return "🏆";
    if (rank <= 50) return "⭐";
    return "📊";
  };

  return (
    <div className={`bg-white rounded-xl shadow-lg border border-gray-200 overflow-hidden ${className}`}>
      {/* 表格標題 */}
      <div className="px-6 py-4 bg-gradient-to-r from-gray-50 to-gray-100 border-b">
        <div className="flex justify-between items-center">
          <h3 className="text-lg font-semibold text-gray-900">
            🏆 持有者排行榜
          </h3>
          <div className="text-sm text-gray-600">
            顯示 {holders.length} 個持有者
          </div>
        </div>
      </div>

      {/* 桌面版表格 */}
      <div className="hidden lg:block overflow-x-auto">
        <table className="w-full">
          <thead className="bg-gray-50 border-b">
            <tr>
              <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                <SortButton field="rank">排名</SortButton>
              </th>
              <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                <SortButton field="address">持有者地址</SortButton>
              </th>
              <th className="px-6 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">
                <SortButton field="balanceNumber">持有數量</SortButton>
              </th>
              <th className="px-6 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">
                <SortButton field="percentage">佔比</SortButton>
              </th>
              <th className="px-6 py-3 text-center text-xs font-medium text-gray-500 uppercase tracking-wider">
                操作
              </th>
            </tr>
          </thead>
          <tbody className="divide-y divide-gray-200">
            {holders.map((holder, index) => {
              const globalRank = getGlobalRank(index);
              const isSelected = selectedHolder === holder.address;
              
              return (
                <tr 
                  key={holder.address}
                  className={`
                    hover:bg-gray-50 transition-colors cursor-pointer
                    ${isSelected ? 'bg-blue-50 ring-2 ring-blue-500 ring-inset' : ''}
                  `}
                  onClick={() => {
                    setSelectedHolder(isSelected ? null : holder.address);
                    onHolderClick?.(holder);
                  }}
                >
                  {/* 排名 */}
                  <td className="px-6 py-4 whitespace-nowrap">
                    <div className="flex items-center gap-3">
                      <div className={`
                        w-10 h-10 rounded-full flex items-center justify-center text-sm font-bold
                        ${getRankStyle(globalRank)}
                      `}>
                        <span className="text-xs">{getRankIcon(globalRank)}</span>
                      </div>
                      <div>
                        <div className="text-sm font-semibold text-gray-900">#{globalRank}</div>
                        {globalRank <= 10 && (
                          <div className="text-xs text-gray-500">前十名</div>
                        )}
                      </div>
                    </div>
                  </td>

                  {/* 地址 */}
                  <td className="px-6 py-4 whitespace-nowrap">
                    <div className="flex items-center gap-3">
                      <Address 
                        address={holder.address}
                        format="short"
                        size="base"
                        blockExplorerAddressLink={true}
                      />
                      <button
                        onClick={(e) => {
                          e.stopPropagation();
                          copyAddress(holder.address);
                        }}
                        className="p-1 hover:bg-gray-200 rounded transition-colors"
                        title="複製地址"
                      >
                        <DocumentDuplicateIcon className="w-4 h-4 text-gray-500" />
                      </button>
                      {copiedAddress === holder.address && (
                        <span className="text-xs text-green-600 font-medium">已複製!</span>
                      )}
                    </div>
                  </td>

                  {/* 持有數量 */}
                  <td className="px-6 py-4 whitespace-nowrap text-right">
                    <div>
                      <div className="text-sm font-semibold text-gray-900">
                        {Number(holder.balanceFormatted).toLocaleString()} {tokenSymbol}
                      </div>
                      <div className="text-xs text-gray-500">
                        {holder.balanceNumber >= 1000000 
                          ? `${(holder.balanceNumber / 1000000).toFixed(2)}M`
                          : holder.balanceNumber >= 1000
                          ? `${(holder.balanceNumber / 1000).toFixed(2)}K`
                          : holder.balanceNumber.toFixed(2)
                        }
                      </div>
                    </div>
                  </td>

                  {/* 佔比 */}
                  <td className="px-6 py-4 whitespace-nowrap text-right">
                    <div>
                      <div className="text-sm font-semibold text-gray-900">
                        {holder.percentage.toFixed(4)}%
                      </div>
                      <div className="w-20 bg-gray-200 rounded-full h-2 ml-auto mt-1">
                        <div 
                          className="bg-gradient-to-r from-blue-500 to-purple-600 h-2 rounded-full transition-all duration-500"
                          style={{ width: `${Math.min(holder.percentage * 2, 100)}%` }}
                        />
                      </div>
                    </div>
                  </td>

                  {/* 操作 */}
                  <td className="px-6 py-4 whitespace-nowrap text-center">
                    <div className="flex items-center justify-center gap-2">
                      <button
                        onClick={(e) => {
                          e.stopPropagation();
                          // 這裡可以添加查看詳情的邏輯

                        className="p-1 hover:bg-gray-200 rounded transition-colors"
                        title="查看詳情"
                      >
                        <EyeIcon className="w-4 h-4 text-gray-500" />

                        onClick={(e) => {
                          e.stopPropagation();
                          shareHolder(holder);
                        }}
                        className="p-1 hover:bg-gray-200 rounded transition-colors"
                        title="分享"
                      >
                        <ShareIcon className="w-4 h-4 text-gray-500" />
                      </button>
                    </div>
                  </td>
                </tr>
              );
            })}
          </tbody>
        </table>
      </div>

      {/* 移動版卡片佈局 */}
      <div className="lg:hidden divide-y divide-gray-200">
        {holders.map((holder, index) => {
          const globalRank = getGlobalRank(index);
          const isSelected = selectedHolder === holder.address;
          
          return (
            <div 
              key={holder.address}
              className={`
                p-4 hover:bg-gray-50 transition-colors cursor-pointer
                ${isSelected ? 'bg-blue-50 ring-2 ring-blue-500 ring-inset' : ''}
              `}
              onClick={() => {
                setSelectedHolder(isSelected ? null : holder.address);
                onHolderClick?.(holder);
              }}
            >
              <div className="flex items-center justify-between mb-3">
                <div className="flex items-center gap-3">
                  <div className={`
                    w-10 h-10 rounded-full flex items-center justify-center text-sm font-bold
                    ${getRankStyle(globalRank)}
                  `}>
                    <span className="text-xs">{getRankIcon(globalRank)}</span>
                  </div>
                  <div>
                    <div className="text-sm font-semibold text-gray-900">#{globalRank}</div>
                    <div className="text-xs text-gray-500">排名</div>
                  </div>
                </div>
                <div className="flex gap-2">
                  <button
                    onClick={(e) => {
                      e.stopPropagation();
                      copyAddress(holder.address);
                    }}
                    className="p-2 hover:bg-gray-200 rounded-full transition-colors"

                    <DocumentDuplicateIcon className="w-4 h-4 text-gray-500" />

                  <button

                      shareHolder(holder);

                    format="short"

                    <div className="text-xs text-gray-500 mb-1">持有數量</div>
                    <div className="text-sm font-semibold text-gray-900">

                    </div>
                  </div>
                  <div>
                    <div className="text-xs text-gray-500 mb-1">佔比</div>
                    <div className="text-sm font-semibold text-gray-900">
                      {holder.percentage.toFixed(4)}%
                    </div>
                  </div>
                </div>

                <div className="w-full bg-gray-200 rounded-full h-2 mt-2">
                  <div 
                    className="bg-gradient-to-r from-blue-500 to-purple-600 h-2 rounded-full transition-all duration-500"
                    style={{ width: `${Math.min(holder.percentage * 2, 100)}%` }}
                  />
                </div>
              </div>
            </div>
          );
        })}
      </div>

      {/* 空狀態 */}
      {holders.length === 0 && (
        <div className="text-center py-12">
          <div className="text-4xl mb-4">🔍</div>
          <h4 className="text-lg font-medium text-gray-900 mb-2">沒有找到持有者</h4>
          <p className="text-gray-500">嘗試調整搜尋條件或清除篩選器</p>

      )}
```
